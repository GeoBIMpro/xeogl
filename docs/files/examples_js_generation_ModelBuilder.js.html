File: examples/js/generation/ModelBuilder.js

/**
 A **ModelBuilder** is a {{#crossLink "Model"}}{{/crossLink}} that provides a "stateful builder" API through which you can
 procedurally generate xeogl content.

 <a href="../../examples/#models_ModelBuilder_city"><img src="http://i.giphy.com/l0HlPJO1AN01Lz27e.gif"></img></a>

 ## Overview

 * A BuilderModel implements the [Builder pattern](https://en.wikipedia.org/wiki/Builder_pattern).
 * Create various assets within a BuilderModel, such as {{#crossLink "Geometry"}}Geometries{{/crossLink}}
 and {{#crossLink "Material"}}Materials{{/crossLink}}, then create {{#crossLink "Entity"}}Entities{{/crossLink}} that use those assets.
 * The BuilderModel then owns those components and will destroy them when you
 call its {{#crossLink "ModelBuilder/clear:method"}}clear(){{/crossLink}} or {{#crossLink "Component/destroy:method"}}destroy(){{/crossLink}} methods.
 * A ModelBuilder can be transformed within World space by attaching it to a {{#crossLink "Transform"}}{{/crossLink}}.
 * A ModelBuilder provides its World-space boundary as a {{#crossLink "Boundary3D"}}{{/crossLink}}.

 ## Examples

 * [Generating a city with a ModelBuilder](../../examples/#models_ModelBuilder_city)</li>

 ## Usage

 A ModelBuilder containing ten textured boxes with random sizes and positions:

 ````javascript
 var model = new xeogl.ModelBuilder();

 // Add a BoxGeometry asset
 ModelBuilder.asset("boxGeometry", {
     type: "xeogl.BoxGeometry"
 });

 // Add a PhongMaterial asset
 ModelBuilder.asset("gridMaterial", {
     type: "xeogl.PhongMaterial",
     ambient: [0.9, 0.3, 0.9],
     shininess: 30,
     diffuseMap: {
         src: "textures/diffuse/gridMaterial.jpg"
     }
 });

 // Set the BoxGeometry asset as the current geometry
 ModelBuilder.geometry("boxGeometry");

 // Set the PhongMaterial asset as the current material
 ModelBuilder.material("gridMaterial");

 // Build ten entities with random sizes and positions,
 // that each get the current geometry and material
 for (var i = 0; i < 10; i++) {

     ModelBuilder.scale(Math.random() * 10 + 1, Math.random() * 10 + 1, Math.random() * 10 + 1);
     ModelBuilder.pos(Math.random() * 100 - 50, Math.random() * 100 - 50, Math.random() * 100 - 50);

     ModelBuilder.entity();
 }
 ````


 @constructor
 @param [scene] {Scene} Parent {{#crossLink "Scene"}}Scene{{/crossLink}} - creates this ModelBuilder in the default
 {{#crossLink "Scene"}}Scene{{/crossLink}} when omitted.
 @param [cfg] {*} Configs
 @param [cfg.id] {String} Optional ID, unique among all components in the parent {{#crossLink "Scene"}}Scene{{/crossLink}},
 generated automatically when omitted.
 @param [cfg.meta] {String:Object} Optional map of user-defined metadata to attach to this ModelBuilder.
 @param [cfg.transform] {Number|String|Transform} A Local-to-World-space (modelling) {{#crossLink "Transform"}}{{/crossLink}} to attach to this ModelBuilder.
 Must be within the same {{#crossLink "Scene"}}{{/crossLink}} as this ModelBuilder. Internally, the given
 {{#crossLink "Transform"}}{{/crossLink}} will be inserted above each top-most {{#crossLink "Transform"}}Transform{{/crossLink}}
 that the ModelBuilder attaches to its {{#crossLink "Entity"}}Entities{{/crossLink}}.
 @extends Model
 */
(function () {

    "use strict";

    xeogl.ModelBuilder = xeogl.Component.extend({

        type: "xeogl.ModelBuilder",

        _init: function (cfg) {

            this._super(cfg);

            this._positions = undefined;
            this._normals = undefined;
            this._uv = undefined;
            this._colors = undefined;
            this._indices = undefined;

            this._matrix = undefined;

            this._needNewMesh = true;
            this._needNewEntity = true;
            this._needNewModel = true;
        },

        _props: {

            model: {

                set: function (model) {
                    this._model = model;
                },

                get: function () {
                    return this._model;
                }
            },

            geometry: {

                set: function (geometry) {
                    this._geometry = geometry;
                },

                get: function () {
                    return this._geometry;
                }
            },

            material: {

                set: function (material) {
                    this._material = material;
                },

                get: function () {
                    return this._material;
                }
            },

            matrix: {

                set: function (matrix) {
                    this._matrix = matrix;
                    return this;
                },

                get: function () {
                    return this._matrix;
                }
            }
        },

        buildMesh: (function () {

            var math = xeogl.math;

            var tempVec3a = math.vec3();
            var tempVec3b = math.vec3();

            return function () {

                var i;
                var len;
                var indicesBump = (this._positions) ? (this._positions.length / 3) : 0;

                if (this._geometry.positions) {

                    if (!this._positions) {
                        this._positions = [];
                    }

                    var positions = this._geometry.positions;

                    if (!this._matrix) {

                        for (i = 0, len = positions.length; i < len; i++) {
                            this._positions.push(positions[i]);
                        }

                    } else {

                        for (i = 0, len = positions.length; i < len; i += 3) {

                            tempVec3a[0] = positions[i + 0];
                            tempVec3a[1] = positions[i + 1];
                            tempVec3a[2] = positions[i + 2];

                            math.transformPoint3(this._matrix, tempVec3a, tempVec3b);

                            this._positions.push(tempVec3b[0]);
                            this._positions.push(tempVec3b[1]);
                            this._positions.push(tempVec3b[2]);
                        }
                    }
                }

                if (this._geometry.normals) {
                    if (!this._normals) {
                        this._normals = [];
                    }
                    for (i = 0, len = this._geometry.normals.length; i < len; i++) {
                        this._normals.push(this._geometry.normals[i]);
                    }
                }

                if (this._geometry.uv) {
                    if (!this._uv) {
                        this._uv = [];
                    }
                    for (i = 0, len = this._geometry.uv.length; i < len; i++) {
                        this._uv.push(this._geometry.uv[i]);
                    }
                }

                if (this._geometry.colors) {
                    if (!this._colors) {
                        this._colors = [];
                    }
                    for (i = 0, len = this._geometry.colors.length; i < len; i++) {
                        this._colors.push(this._geometry.colors[i]);
                    }
                }

                if (this._geometry.indices) {
                    if (!this._indices) {
                        this._indices = [];
                    }
                    for (i = 0, len = this._geometry.indices.length; i < len; i++) {
                        this._indices.push(this._geometry.indices[i] + indicesBump);
                    }
                }

                return this;
            };
        })(),

        buildEntity: function () {

            this._model.create({
                type: "xeogl.Entity",
                geometry: this._model.create({
                    type: "xeogl.Geometry",
                    primitive: this.primitive || "triangles",
                    positions: this._positions,
                    normals: this._normals,
                    uv: this._uv,
                    colors: this._colors,
                    indices: this._indices
                }),
                material: this._material
            });

            this._positions = null;
            this._normals = null;
            this._uv = null;
            this._colors = null;
            this._indices = null;
        }
    });
})();
